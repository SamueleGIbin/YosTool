        const origOpen = XMLHttpRequest.prototype.open;
        const origSend = XMLHttpRequest.prototype.send;

            this._tntUrl = url;
            return origOpen.apply(this, arguments);
        };
        XMLHttpRequest.prototype.send = function (body) {
            this.addEventListener("load", function () {
                try {
                    if (this._tntUrl && String(this._tntUrl).indexOf("dockingDoors") !== -1) {

                        return;

                    const dati = JSON.parse(this.responseText);
                    if (sembraListaMovimenti(dati)) elaboreRispostaMovimenti(dati);
                } catch (e) {}
            });
            return origSend.apply(this, arguments);

    const cacheMovimentiPerUnit = new Map();

    function sembraListaMovimenti(dati) {
        return Array.isArray(dati) && dati.length > 0 && dati[0] && typeof dati[0] === "object"
            && "movementId" in dati[0] && Array.isArray(dati[0].units) && "origin" in dati[0];
    }

    function elaboreRispostaMovimenti(movimenti) {
        movimenti.forEach((m) => {
            if (!m || !Array.isArray(m.units)) return;
            m.units.forEach((u) => {
                if (!u || !u.unitId) return;
                cacheMovimentiPerUnit.set(String(u.unitId), {
                    origin: m.origin,
                    destination: m.destination,
                    scheduleArrival: m.scheduleArrival,
                    estimatedArrival: m.estimatedArrival,
                    actualArrival: m.actualArrival,
                    movementName: m.movementName,
                    tripTourDescription: m.trip ? m.trip.tripTourDescription : ""
                });
            });
        });
    }

    function calcolaCutoffDaCacheMovimenti(doorNumber) {
        const arrivi = caricaArriviBaia();
        const voce = arrivi[doorNumber];
        if (!voce || !voce.unitId) return null;
        const mov = cacheMovimentiPerUnit.get(String(voce.unitId));
        if (!mov || !mov.origin) return null;

        // Preferiamo l'arrivo reale (actualArrival) allo Sched./Estimated Arrival per stabilire
        // la categoria/cutoff di competenza: usare solo l'orario schedulato puo' produrre cutoff
        // gia' passati (es. codici con cutoff nelle prime ore del mattino il cui Sched. Arrival
        // era gia' nelle prime ore dello stesso giorno).
        const arrivoIso = mov.actualArrival || mov.scheduleArrival || mov.estimatedArrival;
        if (!arrivoIso) return null;
        const arrivo = new Date(arrivoIso);
        if (isNaN(arrivo.getTime())) return null;

        // Orario di arrivo REALE alla porta (quando il camion e' stato fisicamente agganciato
        // alla baia): e' il riferimento usato per la regola dei 150 minuti di lavorazione.
        let arrivoPortaReale = null;
        if (voce.orario) {
            const d = new Date(voce.orario);
            if (!isNaN(d.getTime())) arrivoPortaReale = d;
        }

        const risultato = calcolaCutoffPerCodice(mov.origin, arrivo, arrivoPortaReale);
        if (!risultato) return null;
        return { cutoff: risultato, codice: mov.origin };
    }

    function elaboraRispostaDockingDoors(porte) {
        if (!Array.isArray(porte)) return;
        const arrivi = caricaArriviBaia();
        let modificato = false;
        porte.forEach((porta) => {
            const doorNumber = porta && porta.doorName;
            if (!doorNumber) return;
            const unit = porta.unitDTO;
            if (porta.status !== "OCCUPIED" || !unit) {
                if (arrivi[doorNumber]) {
                    delete arrivi[doorNumber];
                    modificato = true;
                }
                return;
            }
            const unitId = unit.unitId;
            const esistente = arrivi[doorNumber];
            if (!esistente || esistente.unitId !== unitId || unit.currentAction === "START_UNLOADING") {
                let orario;
                if (unit.currentAction === "START_UNLOADING" && (unit.updatedDateTimestamp || unit.updatedDate)) {
                    orario = unit.updatedDateTimestamp || unit.updatedDate;
                } else if (unit.currentAction === "LOCATION_ASSIGNED" && (unit.updatedDateTimestamp || unit.updatedDate)) {
                    orario = unit.updatedDateTimestamp || unit.updatedDate;
                } else {
                    orario = new Date().toISOString();
                }
                arrivi[doorNumber] = { unitId: unitId, orario: orario };
                modificato = true;

                registraInStoricoReport(doorNumber, unitId, orario);
            }
        });
        if (modificato) salvaArriviBaia(arrivi);
    }

    function categoriaBaia(door) {
        if (/^P/i.test(door)) return "Parcheggio";
        const n = parseInt(door, 10);
        if (!isNaN(n)) {
            const inScarico = (n >= 101 && n <= 107) ||
                (n >= 10 && n <= 14) || (n >= 201 && n <= 207) ||
                (n >= 601 && n <= 609) || (n >= 501 && n <= 509);
            if (inScarico) return "Scarico";
            if (n >= 301 && n <= 456) return "Carico";
        }
        return "Freight";
    }

    // Costruisce, a partire da un elenco di categorie, la lista di tutti gli orari di cutoff
    // distinti definiti nel sistema (indipendentemente dai codici). Serve per trovare il
    // "prossimo cutoff utile" quando un camion arrives troppo tardi per quello naturalmente
    // assegnato al proprio codice.
    function elencoSlotCutoffUnici() {
        const visti = new Set();
        const slot = [];
        CATEGORIE.forEach((c) => {
            const chiave = c.cutoffOre + ":" + c.cutoffMin;
            if (!visti.has(chiave)) {
                visti.add(chiave);
                slot.push({ ore: c.cutoffOre, min: c.cutoffMin });
            }
        });
        return slot;
    }
    const SLOT_CUTOFF_UNICI = elencoSlotCutoffUnici();

    // Trova il primo cutoff (tra TUTTI quelli definiti nel sistema, non solo quelli del codice)
    // successivo a "cutoffBase" che garantisce almeno 150 minuti di margine rispetto
    // all'orario di arrivo reale alla baia. Restituisce null se non ne trova uno entro 2 giorni.
    function trovaProssimoCutoffConMargine(cutoffBase, arrivoPortaReale) {
        const candidati = [];
        for (let offset = 0; offset <= 2; offset++) {
            SLOT_CUTOFF_UNICI.forEach((slot) => {
                const d = new Date(cutoffBase.getFullYear(), cutoffBase.getMonth(), cutoffBase.getDate() + offset, slot.ore, slot.min, 0, 0);
                if (d.getTime() > cutoffBase.getTime()) candidati.push(d);
            });
        }
        candidati.sort((a, b) => a.getTime() - b.getTime());
        for (const d of candidati) {
            const margine = (d.getTime() - arrivoPortaReale.getTime()) / 60000;
            if (margine >= 150) return d;
        }
        return candidati.length > 0 ? candidati[candidati.length - 1] : null;
    }

    function calcolaCutoffPerCodice(codice, arrivo, arrivoPortaReale) {
        const candidati = CATEGORIE.filter((c) => c.codici.includes(codice));
        if (candidati.length === 0) return null;

        // Per ogni categoria che comprende questo codice, prendiamo SOLO la prima
        // occorrenza futura del suo cutoff (non tutte le ricorrenze sui giorni successivi).
        const choices = [];
        candidati.forEach((cat) => {
            for (let offset = -1; offset <= 2; offset++) {
                const d = new Date(arrivo.getFullYear(), arrivo.getMonth(), arrivo.getDate() + offset, cat.cutoffOre, cat.cutoffMin, 0, 0);
                if (d.getTime() >= arrivo.getTime()) {
                    choices.push({ cat: cat, cutoffDate: d });
                    break;
                }
            }
        });
        if (choices.length === 0) return null;
        choices.sort((a, b) => a.cutoffDate.getTime() - b.cutoffDate.getTime());

        let risultato = null;
        if (choices.length > 1) {
            for (let i = 0; i < choices.length; i++) {
                const diffMinutiArrivoCutoff = (choices[i].cutoffDate.getTime() - arrivo.getTime()) / 60000;
                if (diffMinutiArrivoCutoff >= 150) { risultato = choices[i]; break; }
            }
            if (!risultato) risultato = choices[0];
        } else {
            risultato = choices[0];
        }

        // Regola dei 150 minuti di lavorazione sull'arrivo REALE alla baia: se il camion e'
        // gia' fisicamente arrivato alla porta e tra quell'orario e il cutoff assegnato non ci
        // sono almeno 150 minuti (oppure il cutoff assegnato e' gia' passato), il camion slitta
        // al primo cutoff successivo - tra TUTTI quelli del sistema - che garantisce il margine.
        if (arrivoPortaReale && !isNaN(arrivoPortaReale.getTime())) {
            const margineReale = (risultato.cutoffDate.getTime() - arrivoPortaReale.getTime()) / 60000;
            if (margineReale < 150) {
                const nuovoCutoff = trovaProssimoCutoffConMargine(risultato.cutoffDate, arrivoPortaReale);
                if (nuovoCutoff) {
                    risultato = { cat: risultato.cat, cutoffDate: nuovoCutoff, cutoffOriginale: risultato.cutoffDate, ritardo: true };
                }
            }
        }
        return risultato;
    }

    const TARGET_SELECTOR = ".door-info__movmt-value, .inbound-origin-en-route, .inbound-origin";
    const CODE_ATTR = "data-tnt-code";

    function traduciBlocco(el) {
        let textNode = null;
        for (const child of el.childNodes) {
            if (child.nodeType === 3 && child.nodeValue.trim() !== "") {
                textNode = child;
                break;
            }
        }
        if (!textNode) return;
        const code = textNode.nodeValue.trim();
        if (el.getAttribute(CODE_ATTR) === code) return;
        el.setAttribute(CODE_ATTR, code);
        const vecchiaTraduzione = el.querySelector(".tnt-translation");
        if (vecchiaTraduzione) vecchiaTraduzione.remove();
        const parola = codeMap[code];
        if (!parola) return;
        const span = document.createElement("span");
        span.className = "tnt-translation";
        span.textContent = " (" + parola + ")";
        span.style.cssText = "color:#FF7A00; font-weight:bold; font-family:Roboto,sans-serif; font-size:inherit; white-space:nowrap;";
        el.appendChild(span);
    }

    function scanBlocchi(root) {
        if (root.nodeType !== 1) return;
        if (root.matches && root.matches(TARGET_SELECTOR)) traduciBlocco(root);
        if (root.querySelectorAll) root.querySelectorAll(TARGET_SELECTOR).forEach(traduciBlocco);
    }

    function trovaValorePerChiave(root, chiave) {
        const keys = root.querySelectorAll(".door-info__movmt-key");
        for (const k of keys) {
            if (k.textContent.trim() === chiave) {
                const col = k.closest(".col-4, .col-6");
                return col ? col.querySelector(".door-info__movmt-value") : null;
            }
        }
        return null;
    }

    function estraiCodiceGrezzo(valueEl) {
        if (!valueEl) return null;
        if (valueEl.dataset && valueEl.dataset.tntCode) return valueEl.dataset.tntCode.trim();
        const clone = valueEl.cloneNode(true);
        const trad = clone.querySelector(".tnt-translation");
        if (trad) trad.remove();
        return clone.textContent.trim();
    }

    function parseDataOraSlash(testo) {
        const m = testo.trim().match(/(\d{2})[\/\-](\d{2})\s+(\d{2}):(\d{2})/);
        if (!m) return null;
        const giorno = parseInt(m[1], 10), mese = parseInt(m[2], 10), ore = parseInt(m[3], 10), min = parseInt(m[4], 10);
        return new Date(new Date().getFullYear(), mese - 1, giorno, ore, min);
    }

                            const arriviSalvati = caricaArriviBaia();
                            const voceBaia = arriviSalvati[baiaSelezionata];
                            if (voceBaia && voceBaia.orario) {
                                const d = new Date(voceBaia.orario);
                                if (!isNaN(d.getTime())) { arrivoPortaReale = d; testoArrivoPorta = voceBaia.orario; }
                            }
                        }
                        if (arrivo) risultato = calcolaCutoffPerCodice(codiceOrigine, arrivo, arrivoPortaReale);
                    }
                }
                aggiornaRigaArrivoECountdown(contenitore, risultato);
                if (!origineEl || !arrivoEl) return;
                const chiave = codiceOrigine + "|" + testoArrivo + "|" + testoActualArrivo + "|" + testoArrivoPorta;
                if (sezione.getAttribute("data-tnt-turno-key") === chiave) return;
                sezione.setAttribute("data-tnt-turno-key", chiave);
                const vecchia = sezione.querySelector(".tnt-turno-row");
                if (vecchia) vecchia.remove();
                if (!codiceOrigine || testoArrivo.toUpperCase().includes("N/A")) return;
                const riga = document.createElement("div");
                riga.className = "row door-info__movmt-section-row tnt-turno-row";
                if (risultato) {
                    riga.appendChild(creaColTurno("Turno", risultato.cat.turno));
                    riga.appendChild(creaColTurno("Tipo Cutoff", risultato.cat.nome));
                    const hh = String(risultato.cutoffDate.getHours()).padStart(2, "0");
                    const mm = String(risultato.cutoffDate.getMinutes()).padStart(2, "0");
                    let testoCutoff = hh + ":" + mm;
                    if (risultato.ritardo && risultato.cutoffOriginale) {
                        const hhO = String(risultato.cutoffOriginale.getHours()).padStart(2, "0");
                        const mmO = String(risultato.cutoffOriginale.getMinutes()).padStart(2, "0");
                        testoCutoff = "* " + hhO + ":" + mmO + " -> " + hh + ":" + mm + " per late arrival";
                    }
                    const colCutoff = creaColTurno("Orario Cutoff", testoCutoff, risultato.ritardo ? "col-4" : "col-4");
                    if (risultato.ritardo) {
                        const valEl = colCutoff.querySelector(".door-info__movmt-value");
                        if (valEl) {
                            valEl.style.color = "#ff3b30";
                            valEl.title = "Cutoff slittato: meno di 150 minuti di lavorazione tra l'arrivo reale alla baia e il cutoff originale";
                        }
                    }
                    riga.appendChild(colCutoff);
                } else {
                    riga.appendChild(creaColTurno("Turno", "N/A"));
                    riga.appendChild(creaColTurno("Tipo Cutoff", "N/A"));
                    riga.appendChild(creaColTurno("Orario Cutoff", "N/A"));
                }
                sezione.appendChild(riga);
            } catch (e) {}
        });
    }

    // Colori con cui l'interfaccia YOS marca una unita' come "pronta" (docking indicator blu/verde-acqua).
    const COLORI_UNIT_PRONTA = ["rgb(135,185,172)"];

    // Determina se una riga di outbound e' gia' "pronta" (unit ready) e va quindi esclusa dal menu Carico.
    function rigaSegnalataPronta(riga) {
        if (riga.querySelector(".unit_ready_outline") || riga.classList.contains("unit_ready_outline")) return true;
        const dockingBg = riga.querySelector('[id^="dockingBg_"]');
        if (!dockingBg) return false;
        if (dockingBg.className && typeof dockingBg.className === "string" && /ready/i.test(dockingBg.className)) return true;
        let colore = dockingBg.style && dockingBg.style.backgroundColor ? dockingBg.style.backgroundColor : "";
        if (!colore && window.getComputedStyle) {
            try { colore = window.getComputedStyle(dockingBg).backgroundColor || ""; } catch (e) {}
        }
        colore = colore.replace(/\s+/g, "");
        if (!colore) return false;
        return COLORI_UNIT_PRONTA.some((c) => colore.indexOf(c) !== -1);
    }

    function leggiOutboundInScadenza() {
        const risultati = [];
        const righe = document.querySelectorAll("#outboundListTable tr, table[id*='outbound'] tr, tr.outbound-row");
        righe.forEach((riga) => {
            if (rigaSegnalataPronta(riga)) return;
            const celle = riga.querySelectorAll("td");
            if (celle.length < 2) return;
            let testoOrario = "", testoDestinazione = "";
            celle.forEach((c) => {
                const t = c.textContent.trim();
                if (/^\d{2}:\d{2}$/.test(t) && !testoOrario) testoOrario = t;
                const m = t.match(/\b([A-Z0-9]{3})\b/);
                if (m && codeMap[m[1]] && !testoDestinazione) testoDestinazione = m[1];
            });
            if (!testoOrario || !testoDestinazione) return;
            const parti = testoOrario.split(":");
            const oraTarget = new Date();
            oraTarget.setHours(parseInt(parti[0], 10), parseInt(parti[1], 10), 0, 0);
            const oraCorr = new Date();
            let diffMinuti = (oraTarget.getTime() - oraCorr.getTime()) / 60000;
            if (diffMinuti < -300) diffMinuti += 1440;
            if (diffMinuti <= SOGLIA_OUTBOUND_MINUTI) {
                risultati.push({
                    destinazione: testoDestinazione,
                    nomeEsteso: codeMap[testoDestinazione] || testoDestinazione,
                    orarioStr: testoOrario,
                    minutiMancanti: Math.round(diffMinuti)
                });
            }
        });
        risultati.sort((a, b) => a.minutiMancanti - b.minutiMancanti);
        return risultati;
    }

    function pulisciStoricoVecchio() {
        const storico = caricaStoricoReport();
        if (storico.length === 0) return;
        const limite = Date.now() - GIORNI_CONSERVAZIONE_STORICO * 24 * 60 * 60 * 1000;
        const filtrato = storico.filter((r) => {
            const t = r.timestampIso ? new Date(r.timestampIso).getTime() : NaN;
            return !isNaN(t) && t >= limite;
        });
        if (filtrato.length !== storico.length) salvaStoricoReport(filtrato);
    }

    function registraInStoricoReport(doorNumber, unitId, orarioIso) {
        pulisciStoricoVecchio();
        const storico = caricaStoricoReport();
        const esisteGia = storico.some(item => item.doorNumber === doorNumber && item.unitId === unitId);
        if (esisteGia) return;

        const dataArrivo = new Date(orarioIso);
        const orarioFormattato = String(dataArrivo.getHours()).padStart(2, "0") + ":" + String(dataArrivo.getMinutes()).padStart(2, "0");
        const dataFormattata = String(dataArrivo.getDate()).padStart(2, "0") + "/" + String(dataArrivo.getMonth() + 1).padStart(2, "0") + "/" + dataArrivo.getFullYear();

        const mov = cacheMovimentiPerUnit.get(String(unitId)) || {};
        const codiceOrigine = mov.origin || "N/A";
        const localita = codeMap[codiceOrigine] || codiceOrigine;

        let viaggioStr = "N/A";
        if (mov.tripTourDescription) {
            viaggioStr = mov.tripTourDescription;
        } else if (mov.movementName) {
            viaggioStr = mov.movementName;
        }

        let cutoffOrario = "N/A";
        let cutoffTurno = "N/A";
        let cutoffTipo = "N/A";
        let ritardoFlag = "No";
        let cutoffOriginaleTesto = "";

        const arrivoIso = mov.actualArrival || mov.scheduleArrival || mov.estimatedArrival;
        if (codiceOrigine !== "N/A" && arrivoIso) {
            const arrivoObj = new Date(arrivoIso);
            if (!isNaN(arrivoObj.getTime())) {
                const res = calcolaCutoffPerCodice(codiceOrigine, arrivoObj, dataArrivo);
                if (res) {
                    const hh = String(res.cutoffDate.getHours()).padStart(2, "0");
                    const mm = String(res.cutoffDate.getMinutes()).padStart(2, "0");
                    cutoffOrario = hh + ":" + mm;
                    cutoffTurno = res.cat ? res.cat.turno : "N/A";
                    cutoffTipo = res.cat ? res.cat.nome : "N/A";
                    if (res.ritardo && res.cutoffOriginale) {
                        ritardoFlag = "Si";
                        const hhO = String(res.cutoffOriginale.getHours()).padStart(2, "0");
                        const mmO = String(res.cutoffOriginale.getMinutes()).padStart(2, "0");
                        cutoffOriginaleTesto = hhO + ":" + mmO;
                    }
                }
            }
        }

        storico.push({
            data: dataFormattata,
            doorNumber: doorNumber,
            unitId: unitId,
            orarioArrivo: orarioFormattato,
            codiceOrigine: codiceOrigine,
            localita: localita,
            viaggio: viaggioStr,
            cutoffOrario: cutoffOrario,
            cutoffTurno: cutoffTurno,
            cutoffTipo: cutoffTipo,
            ritardoFlag: ritardoFlag,
            cutoffOriginaleTesto: cutoffOriginaleTesto,
            timestampIso: orarioIso
        });

        salvaStoricoReport(storico);
    }

    function esportaReportCSV(dataInizioStr, dataFineStr) {
        pulisciStoricoVecchio();
        let storico = caricaStoricoReport();

        if (dataInizioStr || dataFineStr) {
            const inizio = dataInizioStr ? new Date(dataInizioStr + "T00:00:00") : null;
            const fine = dataFineStr ? new Date(dataFineStr + "T23:59:59") : null;

            storico = storico.filter((r) => {
                if (!r.timestampIso) return true;
                const d = new Date(r.timestampIso);
                if (isNaN(d.getTime())) return true;
                if (inizio && d < inizio) return false;
                if (fine && d > fine) return false;
                return true;
            });
        }

        if (storico.length === 0) {
            alert("Nessun dato registrato nel periodo selezionato.");
            return;
        }

        let csvContent = "data:text/csv;charset=utf-8,";
        csvContent += "Data;Baia;Unit ID;Ora Arrivo Reale;Codice;Localita;Info Viaggio;Ora Cutoff;Turno;Categoria Cutoff;Slittato per Late Arrival;Cutoff Originale\n";

        storico.forEach(r => {
            const riga = `${r.data};${r.doorNumber};${r.unitId};${r.orarioArrivo};${r.codiceOrigine};${r.localita};${r.viaggio};${r.cutoffOrario};${r.cutoffTurno};${r.cutoffTipo};${r.ritardoFlag || "No"};${r.cutoffOriginaleTesto || ""}`;
            csvContent += riga + "\n";
        });

        const encodedUri = encodeURI(csvContent);
        const link = document.createElement("a");
        link.setAttribute("href", encodedUri);
        const suffisso = (dataInizioStr || dataFineStr) ? `_${dataInizioStr || 'inizio'}_al_${dataFineStr || 'fine'}` : "_totale";
        link.setAttribute("download", `Report_Arrivi_Baie${suffisso}.csv`);
        document.body.appendChild(link);
        link.click();
        document.body.removeChild(link);
    }

    function aggiornaRigaArrivoECountdown(ambito, risultato) {
        const vecchia = ambito.querySelector(".tnt-custom-info-row");
        if (vecchia) vecchia.remove();

        const rigaRemarks = trovaRigaPerChiave(ambito, ["Arrival Remarks", "YOS Remarks"]);
        if (!rigaRemarks) return;

        let valoreArrivo = "N/A";
        if (baiaSelezionata) {
            const arrivi = caricaArriviBaia();
            const voce = arrivi[baiaSelezionata];
            if (voce && voce.orario) {
                const dataArrivo = new Date(voce.orario);
                valoreArrivo = String(dataArrivo.getHours()).padStart(2, "0") + ":" + String(dataArrivo.getMinutes()).padStart(2, "0");
            }
        }

        let valoreCountdown = "N/A";
        if (risultato && risultato.cutoffDate) {
            const diffMs = risultato.cutoffDate.getTime() - new Date().getTime();
            if (diffMs > 0) {
                const diffMinsTot = Math.floor(diffMs / 60000);
                const ore = Math.floor(diffMinsTot / 60);
                const minuti = diffMinsTot % 60;
                valoreCountdown = ore > 0 ? `${ore}h ${minuti}m` : `${minuti}m`;
            } else {
                const diffMinsTot = Math.floor(Math.abs(diffMs) / 60000);
                const ore = Math.floor(diffMinsTot / 60);
                const minuti = diffMinsTot % 60;
                valoreCountdown = `In ritardo di ${ore > 0 ? ore + 'h ' : ''}${minuti}m`;
            }
        }

        const rigaNuova = document.createElement("div");
        rigaNuova.className = "row door-info__movmt-section-row tnt-custom-info-row";
        rigaNuova.appendChild(creaColCustom("Ora Arrivo Reale", valoreArrivo));
        rigaNuova.appendChild(creaColCustom("Tempo Rimasto", valoreCountdown));
        rigaRemarks.parentNode.insertBefore(rigaNuova, rigaRemarks.nextSibling);
    }

    function trovaCutoffBaia(doorEl) {
        if (!doorEl) return null;
        const box = doorEl.closest(".door_box_occupied, .door_box, .west_door_occupied, .east_door_occupied, [class*='door_box']");
        if (!box) return null;
        const valueEl = box.querySelector(".door-info__movmt-value, .inbound-origin-en-route, .inbound-origin");
        const codiceOrigine = estraiCodiceGrezzo(valueEl);
        if (!codiceOrigine) return null;

        const schedEl = box.querySelector(".inbound-scheduled-time, [class*='scheduled']");
        let arrivo = null;
        if (schedEl) {
            const t = schedEl.textContent.trim();
            arrivo = parseDataOraSlash(t);
        }
        if (!arrivo) arrivo = new Date();

        const doorNumberTesto = doorEl.textContent.trim();
        let arrivoPortaReale = null;
        if (doorNumberTesto) {
            const arriviSalvati = caricaArriviBaia();
            const voceBaia = arriviSalvati[doorNumberTesto];
            if (voceBaia && voceBaia.orario) {
                const d = new Date(voceBaia.orario);
                if (!isNaN(d.getTime())) arrivoPortaReale = d;
            }
        }

        const cutoffRes = calcolaCutoffPerCodice(codiceOrigine, arrivo, arrivoPortaReale);
        if (cutoffRes) return { cutoff: cutoffRes, codice: codiceOrigine };
        return null;
    }

    function scansionaBaieScarico(panelBody) {
        const gruppi = new Map();
        const ora = new Date();
        let trovatoQualcosa = false;

        function aggiungiAlGruppo(risultato, infoBaia) {
            const diffMinuti = (risultato.cutoffDate.getTime() - ora.getTime()) / 60000;
            if (diffMinuti > SOGLIA_SCARICO_MINUTI) return;

            const hh = String(risultato.cutoffDate.getHours()).padStart(2, "0");
            const mm = String(risultato.cutoffDate.getMinutes()).padStart(2, "0");
            const chiaveGruppo = `${risultato.cat.nome}|${hh}:${mm}`;

            if (!gruppi.has(chiaveGruppo)) {
                gruppi.set(chiaveGruppo, {
                    nomeCategoria: risultato.cat.nome,
                    orarioStr: `${hh}:${mm}`,
                    diffMinuti: diffMinuti,
                    baie: []
                });
            }
            gruppi.get(chiaveGruppo).baie.push(infoBaia);
            trovatoQualcosa = true;
        }

        const porteDom = document.querySelectorAll(".door_number, .west_door_number, .east_door_number, .reverse_door_number");
        porteDom.forEach((doorEl) => {
            const doorNumber = doorEl.textContent.trim();
            if (categoriaBaia(doorNumber) !== "Scarico") return;
            const res = trovaCutoffBaia(doorEl);
            if (!res) return;
            const nomeEsteso = codeMap[res.codice] || res.codice;
            aggiungiAlGruppo(res.cutoff, `${doorNumber} (${nomeEsteso})`);
        });

        const arrivi = caricaArriviBaia();
        Object.keys(arrivi).forEach((doorNumber) => {
            if (categoriaBaia(doorNumber) !== "Scarico") return;
            const res = calcolaCutoffDaCacheMovimenti(doorNumber);
            if (!res) return;
            const giaPresente = Array.from(porteDom).some((el) => el.textContent.trim() === doorNumber);
            if (giaPresente) return;
            const nomeEsteso = codeMap[res.codice] || res.codice;
            aggiungiAlGruppo(res.cutoff, `${doorNumber} (${nomeEsteso})`);
        });

        if (!trovatoQualcosa) {
            panelBody.innerHTML = '<div class="tnt-empty">Nessun orario di scarico in scadenza nelle baie.</div>';
            return;
        }

        const listaGruppi = Array.from(gruppi.values());
        listaGruppi.sort((a, b) => a.diffMinuti - b.diffMinuti);

        listaGruppi.forEach((g) => {
            const minMancanti = Math.round(g.diffMinuti);
            let classeCard = "tnt-card-verde";
            let testoBadge = "";

            if (minMancanti < 0) {
                classeCard = "tnt-card-rosso";
                const rit = Math.abs(minMancanti);
                const oreRit = Math.floor(rit / 60);
                const minRit = rit % 60;
                testoBadge = `In ritardo di ${oreRit > 0 ? oreRit + 'h ' : ''}${minRit}m`;
            } else if (minMancanti <= 30) {
                classeCard = "tnt-card-rosso";
                testoBadge = `${minMancanti} min`;
            } else if (minMancanti <= 60) {
                classeCard = "tnt-card-ambra";
                testoBadge = `${minMancanti} min`;
            } else {
                const ore = Math.floor(minMancanti / 60);
                const min = minMancanti % 60;
                testoBadge = ore > 0 ? `${ore}h ${min}m` : `${min}m`;
            }

            const card = document.createElement("div");
            card.className = `tnt-card ${classeCard}`;
            card.innerHTML = `
                <div class="tnt-card-row">
                    <span class="tnt-card-title">${g.nomeCategoria} (${g.orarioStr})</span>
                    <span class="tnt-badge">${testoBadge}</span>
                </div>
                <div class="tnt-card-sub">Baie: ${g.baie.join(", ")}</div>
            `;
            panelBody.appendChild(card);
        });
    }

    function creaMenuAffiancato(idBtn, testoBtn, larghezzaPannello, popolaCallback, haPallino = false) {
        const container = document.createElement("li");
        container.style.cssText = "position:relative; display:inline-block;";

        const btn = document.createElement("a");
        btn.id = idBtn;
        btn.href = "javascript:void(0);";

        let dotSpan = null;
        if (haPallino) {
            dotSpan = document.createElement("span");
            dotSpan.style.display = "none";
            btn.appendChild(dotSpan);
        }

        const labelText = document.createTextNode(testoBtn);
        btn.appendChild(labelText);

        const panel = document.createElement("div");
        panel.className = "tnt-panel";
        panel.style.width = larghezzaPannello;

        panel.innerHTML = `
            <div class="tnt-panel-header">
                <div class="tnt-panel-title">${testoBtn}</div>
                <div class="tnt-panel-close">&times;</div>
            </div>
            <div class="tnt-panel-body"></div>
        `;

        container.appendChild(btn);
        document.body.appendChild(panel);

        const panelBody = panel.querySelector(".tnt-panel-body");
        const panelClose = panel.querySelector(".tnt-panel-close");

        function posizionaPannello() {
            const rect = btn.getBoundingClientRect();
            panel.style.top = rect.bottom + 4 + "px";
            panel.style.left = rect.left + "px";
            panel.style.maxHeight = `calc(100vh - ${rect.bottom + 20}px)`;
        }

        function apri() {
            document.querySelectorAll(".tnt-panel").forEach((p) => { if (p !== panel) p.style.display = "none"; });
            posizionaPannello();
            panel.style.display = "flex";
            popolaCallback(panelBody);
        }

        function chiudi() { panel.style.display = "none"; }

        btn.addEventListener("click", (e) => {
            e.preventDefault();
            e.stopPropagation();
            if (panel.style.display === "flex") chiudi(); else apri();
        });

        panelClose.addEventListener("click", (e) => { e.stopPropagation(); chiudi(); });
        panel.addEventListener("click", (e) => { e.stopPropagation(); });

        document.addEventListener("click", (e) => {
            if (panel.style.display === "flex" && !panel.contains(e.target) && e.target !== btn) chiudi();
        });

        window.addEventListener("resize", () => {
            if (panel.style.display === "flex") posizionaPannello();
        });

        return { container, dotSpan, panel, panelBody };
    }

    function creaMenuReport(idBtn, testoBtn, larghezzaPannello) {
        const container = document.createElement("li");
        container.style.cssText = "position:relative; display:inline-block;";

        const btn = document.createElement("a");
        btn.id = idBtn;
        btn.href = "javascript:void(0);";
        btn.textContent = testoBtn;

        const panel = document.createElement("div");
        panel.className = "tnt-panel";
        panel.style.width = larghezzaPannello;

        const oggiStr = new Date().toISOString().split("T")[0];

        panel.innerHTML = `
            <div class="tnt-panel-header">
                <div class="tnt-panel-title">${testoBtn}</div>
                <div class="tnt-panel-close">&times;</div>
            </div>
            <div class="tnt-panel-body">
                <label>Data Inizio:</label>
                <input type="date" id="tnt_report_inizio" value="${oggiStr}">
                <label>Data Fine:</label>
                <input type="date" id="tnt_report_fine" value="${oggiStr}">
                <button id="tnt_btn_scarica_filtrato" class="tnt-btn-primary">Scarica CSV (Selezionato)</button>
                <button id="tnt_btn_scarica_tutto" class="tnt-btn-secondary">Scarica CSV (Tutto lo storico)</button>
                <div class="tnt-footnote">* Si consiglia di esportare il report ogni giorno prima delle 00:00 per garantire la completezza dei dati registrati durante il turno. I dati registrati vengono comunque conservati per 3 giorni e ripuliti automaticamente.</div>
            </div>
        `;

        container.appendChild(btn);
        document.body.appendChild(panel);

        const panelClose = panel.querySelector(".tnt-panel-close");

        function posizionaPannello() {
            const rect = btn.getBoundingClientRect();
            panel.style.top = rect.bottom + 4 + "px";
            panel.style.left = rect.left + "px";
        }

        function apri() {
            document.querySelectorAll(".tnt-panel").forEach((p) => { if (p !== panel) p.style.display = "none"; });
            posizionaPannello();
            panel.style.display = "flex";
        }

        function chiudi() { panel.style.display = "none"; }

        btn.addEventListener("click", (e) => {
            e.preventDefault();
            e.stopPropagation();
            if (panel.style.display === "flex") chiudi(); else apri();
        });

        panelClose.addEventListener("click", (e) => { e.stopPropagation(); chiudi(); });
        panel.addEventListener("click", (e) => { e.stopPropagation(); });

        document.addEventListener("click", (e) => {
            if (panel.style.display === "flex" && !panel.contains(e.target) && e.target !== btn) chiudi();
        });

        panel.querySelector("#tnt_btn_scarica_filtrato").addEventListener("click", () => {
            const ini = panel.querySelector("#tnt_report_inizio").value;
            const fin = panel.querySelector("#tnt_report_fine").value;
            esportaReportCSV(ini, fin);
            chiudi();
        });

        panel.querySelector("#tnt_btn_scarica_tutto").addEventListener("click", () => {
            esportaReportCSV(null, null);
            chiudi();
        });

        return container;
    }

    let menuScarico = null;
    let menuCarico = null;
    let menuReportObj = null;

    function initMenuHeader() {
        if (document.getElementById("tnt_menu_scarico")) return;
        const ul = document.querySelector(".navbar-nav, .nav.navbar-nav, header ul");
        if (!ul) return;

        menuScarico = creaMenuAffiancato("tnt_menu_scarico", "Scarico", "320px", (body) => {
            body.innerHTML = "";
            scansionaBaieScarico(body);
        });

        menuCarico = creaMenuAffiancato("tnt_menu_carico", "Carico", "320px", (body) => {
            body.innerHTML = "";
            scansionaOutboundCarico(body);
        }, true);

        menuReportObj = creaMenuReport("tnt_menu_report", "Report CSV", "280px");

        ul.appendChild(menuScarico.container);
        ul.appendChild(menuCarico.container);
        ul.appendChild(menuReportObj);
    }

    function aggiornaPannelloOutbound() {
        if (!menuCarico || !menuCarico.dotSpan) return;
        const lista = leggiOutboundInScadenza();
        const minImminente = lista.length > 0 ? lista[0].minutiMancanti : 999;

        if (minImminente <= 15) {
            menuCarico.dotSpan.className = "tnt-dot-rosso";
            menuCarico.dotSpan.style.display = "inline-block";
        } else if (minImminente <= 30) {
            menuCarico.dotSpan.className = "tnt-dot-giallo";
            menuCarico.dotSpan.style.display = "inline-block";
        } else {
            menuCarico.dotSpan.style.display = "none";
        }

        if (menuCarico.panel.style.display === "flex") {
            menuCarico.panelBody.innerHTML = "";
            scansionaOutboundCarico(menuCarico.panelBody);
        }

        if (menuScarico && menuScarico.panel.style.display === "flex") {
            menuScarico.panelBody.innerHTML = "";
            scansionaBaieScarico(menuScarico.panelBody);
        }
    }

    function initIcone() {
        initMenuHeader();
    }

    function aggiornaTracciamentoArrivi() {
        monitoraBaiaSelezionata();
    }

    try { scanBlocchi(document.body); } catch (e) {}
    try { initIcone(); } catch (e) {}
    try { aggiornaTracciamentoArrivi(); } catch (e) {}
    try { processaTripDetails(); } catch (e) {}

    let debounceTimer = null;
    const nodiDaControllare = new Set();
    function eseguiControlliPesanti() {
        const nodi = Array.from(nodiDaControllare);
        nodiDaControllare.clear();
        try { nodi.forEach((n) => { if (document.body.contains(n)) scanBlocchi(n); }); } catch (e) {}
        try { initIcone(); } catch (e) {}
        try { aggiornaTracciamentoArrivi(); } catch (e) {}
        try { processaTripDetails(); } catch (e) {}
        try { aggiornaPannelloOutbound(); } catch (e) {}
    }

    const observer = new MutationObserver((mutations) => {
        for (const m of mutations) {
            if (m.type === "childList") {
                m.addedNodes.forEach((n) => { if (n.nodeType === 1) nodiDaControllare.add(n); });
            } else if (m.type === "characterData" && m.target.parentElement) {
                nodiDaControllare.add(m.target.parentElement);
            } else if (m.type === "attributes" && m.target.nodeType === 1) {
                nodiDaControllare.add(m.target);
            }
        }
        clearTimeout(debounceTimer);
        debounceTimer = setTimeout(eseguiControlliPesanti, 600);
    });
    observer.observe(document.body, { childList: true, subtree: true, attributes: true, characterData: true });

    setInterval(() => {
        try { aggiornaPannelloOutbound(); } catch (e) {}
        try { processaTripDetails(); } catch (e) {}
    }, 15000);

})();

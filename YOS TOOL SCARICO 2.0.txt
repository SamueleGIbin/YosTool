(function () {
    'use strict';

    function gestisciRispostaIntercettata(url, testoBody) {
        const urlStr = String(url || "");
        try {
            if (urlStr.indexOf("dockingDoors") !== -1) {
                elaboraRispostaDockingDoors(JSON.parse(testoBody));
                return;
            }
        } catch (e) {}
        try {
            const dati = JSON.parse(testoBody);
            if (sembraListaMovimenti(dati)) elaboreRispostaMovimenti(dati);
        } catch (e) {}
    }

    (function interceptaXHR() {
        const origOpen = XMLHttpRequest.prototype.open;
        const origSend = XMLHttpRequest.prototype.send;
        XMLHttpRequest.prototype.open = function (method, url) {
            this._tntUrl = url;
            return origOpen.apply(this, arguments);
        };
        XMLHttpRequest.prototype.send = function (body) {
            this.addEventListener("load", function () {
                try { gestisciRispostaIntercettata(this._tntUrl, this.responseText); } catch (e) {}
            });
            return origSend.apply(this, arguments);
        };
    })();

    (function interceptaFetch() {
        if (typeof window.fetch !== "function") return;
        const origFetch = window.fetch;
        window.fetch = function (input, init) {
            const url = typeof input === "string" ? input : (input && input.url) || "";
            return origFetch.apply(this, arguments).then((response) => {
                try {
                    response.clone().text().then((testo) => {
                        try { gestisciRispostaIntercettata(url, testo); } catch (e) {}
                    }).catch(() => {});
                } catch (e) {}
                return response;
            });
        };
    })();

    const cacheMovimentiPerUnit = new Map();
    const cacheCutoffPerBaia = new Map();

    function sembraListaMovimenti(dati) {
        return Array.isArray(dati) && dati.length > 0 && dati[0] && typeof dati[0] === "object"
            && "movementId" in dati[0] && Array.isArray(dati[0].units) && "origin" in dati[0];
    }

    function elaboreRispostaMovimenti(movimenti) {
        movimenti.forEach((m) => {
            if (!m || !Array.isArray(m.units)) return;
            m.units.forEach((u) => {
                if (!u || !u.unitId) return;
                cacheMovimentiPerUnit.set(String(u.unitId), {
                    origin: m.origin,
                    destination: m.destination,
                    scheduleArrival: m.scheduleArrival,
                    estimatedArrival: m.estimatedArrival,
                    actualArrival: m.actualArrival,
                    movementName: m.movementName,
                    tripTourDescription: m.trip ? m.trip.tripTourDescription : "",
                    movementId: m.movementId != null ? String(m.movementId) : null
                });
            });
        });
    }

    function calcolaCutoffDaCacheMovimenti(doorNumber) {
        const arrivi = caricaArriviBaia();
        const voce = arrivi[doorNumber];
        if (!voce || !voce.unitId) return null;
        const mov = cacheMovimentiPerUnit.get(String(voce.unitId));
        if (!mov || !mov.origin) return null;
        const arrivoIso = mov.actualArrival || mov.scheduleArrival || mov.estimatedArrival;
        if (!arrivoIso) return null;
        const arrivo = new Date(arrivoIso);
        if (isNaN(arrivo.getTime())) return null;
        let arrivoPortaReale = null;
        if (voce.orario) {
            const d = new Date(voce.orario);
            if (!isNaN(d.getTime())) arrivoPortaReale = d;
        }
        const risultato = calcolaCutoffPerCodice(mov.origin, arrivo, arrivoPortaReale);
        if (!risultato) return null;
        return { cutoff: risultato, codice: mov.origin };
    }

    function elaboraRispostaDockingDoors(porte) {
        if (!Array.isArray(porte)) return;
        const arrivi = caricaArriviBaia();
        let modificato = false;
        porte.forEach((porta) => {
            const doorNumber = porta && porta.doorName;
            if (!doorNumber) return;
            const unit = porta.unitDTO;
            if (porta.status !== "OCCUPIED" || !unit) {
                if (arrivi[doorNumber]) {
                    delete arrivi[doorNumber];
                    cacheCutoffPerBaia.delete(doorNumber);
                    modificato = true;
                }
                return;
            }
            const unitId = unit.unitId;
            const esistente = arrivi[doorNumber];
            const cambiaUnit = !esistente || esistente.unitId !== unitId;
            if (cambiaUnit || unit.currentAction === "START_UNLOADING") {
                let orario;
                if (unit.currentAction === "START_UNLOADING" && (unit.updatedDateTimestamp || unit.updatedDate)) {
                    orario = unit.updatedDateTimestamp || unit.updatedDate;
                } else if (unit.currentAction === "LOCATION_ASSIGNED" && (unit.updatedDateTimestamp || unit.updatedDate)) {
                    orario = unit.updatedDateTimestamp || unit.updatedDate;
                } else {
                    orario = new Date().toISOString();
                }
                // L'orario reale di arrivo alla baia (voce.orario) è l'input che decide se un
                // camion slitta al cutoff successivo per "late arrival". Se questo orario non
                // era ancora noto la prima volta che è stato calcolato il cutoff (es. Trip
                // Details aperto appena il camion è comparso, prima dello START_UNLOADING),
                // il risultato "non in ritardo" restava in cache per sempre: il pannello
                // laterale non veniva più aggiornato anche quando il camion diventava
                // realmente in ritardo. Invalidiamo la cache ogni volta che l'orario cambia,
                // non solo quando cambia il mezzo.
                const orarioCambiato = !esistente || esistente.orario !== orario;
                if (cambiaUnit || orarioCambiato) {
                    cacheCutoffPerBaia.delete(doorNumber);
                    unitVerificatiCutoff.delete(String(unitId));
                }
                arrivi[doorNumber] = { unitId: unitId, orario: orario };
                modificato = true;

                if (cambiaUnit) registraInStoricoReport(doorNumber, unitId, orario);
                if (categoriaBaia(doorNumber) === "Scarico" && (cambiaUnit || orarioCambiato)) accodaVerificaBaia(doorNumber, unitId);
            }
        });
        if (modificato) salvaArriviBaia(arrivi);
    }

    // Un nuovo truck viene accodato solo nel momento esatto in cui viene rilevato (evento),
    // non con un polling periodico su tutte le baie.
    const unitVerificatiCutoff = new Set();
    const codaVerificaBaie = [];

    function accodaVerificaBaia(doorNumber, unitId) {
        if (!doorNumber || !unitId) return;
        const chiave = String(unitId);
        if (unitVerificatiCutoff.has(chiave)) return;
        if (codaVerificaBaie.some((v) => v.unitId === chiave)) return;
        codaVerificaBaie.push({ doorNumber, unitId: chiave, tentativi: 0 });
    }

    function trovaElementoBaiaPerNumero(doorNumber) {
        const nodi = document.querySelectorAll(DOOR_NUM_SELECTOR_SCARICO);
        for (const numEl of nodi) {
            const span = numEl.querySelector("span");
            if (!span || span.textContent.trim() !== doorNumber) continue;
            const wrapperStandard = numEl.closest('[id^="unit_"]');
            if (wrapperStandard) return wrapperStandard;
            // Alcune baie (es. parcheggi/"reverse") non hanno un antenato con id="unit_...":
            // risaliamo il DOM finché non troviamo un contenitore che include anche il blocco
            // "container_" o "parking_area_" (dove si trova l'icona blu di "pronto").
            let nodo = numEl.parentElement;
            let salti = 0;
            while (nodo && salti < 6) {
                if (nodo.querySelector('[id^="container_"].unit-assigned, [id^="parking_area_"].unit-assigned')) return nodo;
                nodo = nodo.parentElement;
                salti++;
            }
        }
        return null;
    }

    async function elaboraCodaVerificaBaie() {
        if (codaInLavorazione || codaVerificaBaie.length === 0) return;
        codaInLavorazione = true;
        try {
            const voce = codaVerificaBaie.shift();
            const arrivi = caricaArriviBaia();
            const attuale = arrivi[voce.doorNumber];
            // Se nel frattempo il truck non è più in quella baia, la voce non è più valida.
            if (!attuale || String(attuale.unitId) !== voce.unitId) { codaInLavorazione = false; return; }
            // Se nel frattempo l'API dei movimenti ha già fornito i dati, non serve forzare nulla.
            if (cacheCutoffPerBaia.has(voce.doorNumber) || calcolaCutoffDaCacheMovimenti(voce.doorNumber)) {
                unitVerificatiCutoff.add(voce.unitId);
                codaInLavorazione = false;
                return;
            }
            const wrapper = trovaElementoBaiaPerNumero(voce.doorNumber);
            if (!wrapper) { codaVerificaBaie.push(voce); codaInLavorazione = false; return; }
            const baiaPrecedente = baiaSelezionata;
            wrapper.dispatchEvent(new MouseEvent("click", { bubbles: true }));
            await new Promise((r) => setTimeout(r, 900));
            assicuraTabInbound();
            await new Promise((r) => setTimeout(r, 500));
            const info = leggiCutoffBaiaCorrente();
            if (info) {
                cacheCutoffPerBaia.set(voce.doorNumber, info.cutoff);
                unitVerificatiCutoff.add(voce.unitId);
            } else {
                voce.tentativi = (voce.tentativi || 0) + 1;
                if (voce.tentativi < 3) codaVerificaBaie.push(voce);
                else unitVerificatiCutoff.add(voce.unitId);
            }
            if (baiaPrecedente && baiaPrecedente !== voce.doorNumber) {
                const wrapperPrecedente = trovaElementoBaiaPerNumero(baiaPrecedente);
                if (wrapperPrecedente) wrapperPrecedente.dispatchEvent(new MouseEvent("click", { bubbles: true }));
            }
        } catch (e) {}
        codaInLavorazione = false;
    }

    function categoriaBaia(door) {
        if (/^P/i.test(door)) return "Parcheggio";
        const n = parseInt(door, 10);
        if (!isNaN(n)) {
            const inScarico = (n >= 101 && n <= 107) || (n >= 10 && n <= 14) || (n >= 201 && n <= 207) || (n >= 601 && n <= 609) || (n >= 501 && n <= 509);
            if (inScarico) return "Scarico";
            if (n >= 301 && n <= 456) return "Carico";
        }
        return "Freight";
    }

    function elencoSlotCutoffUnici() {
        const visti = new Set();
        const slot = [];
        CATEGORIE.forEach((c) => {
            const chiave = c.cutoffOre + ":" + c.cutoffMin;
            if (!visti.has(chiave)) {
                visti.add(chiave);
                slot.push({ ore: c.cutoffOre, min: c.cutoffMin });
            }
        });
        return slot;
    }
    const SLOT_CUTOFF_UNICI = elencoSlotCutoffUnici();

    function trovaProssimoCutoffConMargine(cutoffBase, arrivoPortaReale) {
        const candidati = [];
        for (let offset = 0; offset <= 2; offset++) {
            SLOT_CUTOFF_UNICI.forEach((slot) => {
                const d = new Date(cutoffBase.getFullYear(), cutoffBase.getMonth(), cutoffBase.getDate() + offset, slot.ore, slot.min, 0, 0);
                if (d.getTime() > cutoffBase.getTime()) candidati.push(d);
            });
        }
        candidati.sort((a, b) => a.getTime() - b.getTime());
        for (const d of candidati) {
            const margine = (d.getTime() - arrivoPortaReale.getTime()) / 60000;
            if (margine >= 150) return d;
        }
        return candidati.length > 0 ? candidati[candidati.length - 1] : null;
    }

    function calcolaCutoffPerCodice(codice, arrivo, arrivoPortaReale) {
        const candidati = CATEGORIE.filter((c) => c.codici.includes(codice));
        if (candidati.length === 0) return null;
        const choices = [];
        candidati.forEach((cat) => {
            for (let offset = -1; offset <= 2; offset++) {
                const d = new Date(arrivo.getFullYear(), arrivo.getMonth(), arrivo.getDate() + offset, cat.cutoffOre, cat.cutoffMin, 0, 0);
                if (d.getTime() >= arrivo.getTime()) {
                    choices.push({ cat: cat, cutoffDate: d });
                    break;
                }
            }
        });
        if (choices.length === 0) return null;
        choices.sort((a, b) => a.cutoffDate.getTime() - b.cutoffDate.getTime());
        let risultato = null;
        if (choices.length > 1) {
            for (let i = 0; i < choices.length; i++) {
                const diffMinutiArrivoCutoff = (choices[i].cutoffDate.getTime() - arrivo.getTime()) / 60000;
                if (diffMinutiArrivoCutoff >= 150) { risultato = choices[i]; break; }
            }
            if (!risultato) risultato = choices[0];
        } else {
            risultato = choices[0];
        }

        if (arrivoPortaReale && !isNaN(arrivoPortaReale.getTime())) {
            const margineReale = (risultato.cutoffDate.getTime() - arrivoPortaReale.getTime()) / 60000;
            if (margineReale < 150) {
                const nuovoCutoff = trovaProssimoCutoffConMargine(risultato.cutoffDate, arrivoPortaReale);
                if (nuovoCutoff) {
                    risultato = { cat: risultato.cat, cutoffDate: nuovoCutoff, cutoffOriginale: risultato.cutoffDate, ritardo: true };
                }
            }
        }
        return risultato;
    }

    const TARGET_SELECTOR = ".door-info__movmt-value, .inbound-origin-en-route, .inbound-origin";
    const CODE_ATTR = "data-tnt-code";
    function traduciBlocco(el) {
        let textNode = null;
        for (const child of el.childNodes) {
            if (child.nodeType === 3 && child.nodeValue.trim() !== "") {
                textNode = child;
                break;
            }
        }
        if (!textNode) return;
        const code = textNode.nodeValue.trim();
        if (el.getAttribute(CODE_ATTR) === code) return;
        el.setAttribute(CODE_ATTR, code);
        const vecchiaTraduzione = el.querySelector(".tnt-translation");
        if (vecchiaTraduzione) vecchiaTraduzione.remove();
        const parola = codeMap[code];
        if (!parola) return;
        const span = document.createElement("span");
        span.className = "tnt-translation";
        span.textContent = " (" + parola + ")";
        span.style.cssText = "color:#FF7A00; font-weight:bold; font-family:Roboto,sans-serif; font-size:inherit; white-space:nowrap;";
        el.appendChild(span);
    }

    function scanBlocchi(root) {
        if (root.nodeType !== 1) return;
        if (root.matches && root.matches(TARGET_SELECTOR)) traduciBlocco(root);
        if (root.querySelectorAll) root.querySelectorAll(TARGET_SELECTOR).forEach(traduciBlocco);
    }

    function estraiCodiceGrezzo(valueEl) {
        if (!valueEl) return null;
        if (valueEl.dataset && valueEl.dataset.tntCode) return valueEl.dataset.tntCode.trim();
        const clone = valueEl.cloneNode(true);
        const trad = clone.querySelector(".tnt-translation");
        if (trad) trad.remove();
        return clone.textContent.trim();
    }

    function parseDataOraSlash(testo) {
        const m = testo.trim().match(/(\d{2})[\/\-](\d{2})\s+(\d{2}):(\d{2})/);
        if (!m) return null;
        const giorno = parseInt(m[1], 10), mese = parseInt(m[2], 10), ore = parseInt(m[3], 10), min = parseInt(m[4], 10);
        return new Date(new Date().getFullYear(), mese - 1, giorno, ore, min);
    }

    function processaTripDetails() {
        const contenitori = document.querySelectorAll(".door-info__movmt-no-left-padding");
        contenitori.forEach((contenitore) => {
            try {
                const tripDetails = contenitore.querySelector(".door-info__trip-details");
                const sezione = contenitore.querySelector(".door-info__movmt-section");
                if (!tripDetails || !sezione) return;
                const selezionato = tripDetails.querySelector(".door-info__header--nav-text--selected");
                if (!selezionato || selezionato.textContent.trim() !== "INBOUND") {
                    const vecchia = sezione.querySelector(".tnt-turno-row");
                    if (vecchia) vecchia.remove();
                    sezione.removeAttribute("data-tnt-turno-key");
                    const vecchiaArrivo = contenitore.querySelector(".tnt-custom-info-row");
                    if (vecchiaArrivo) vecchiaArrivo.remove();
                    return;
                }
                const origineEl = trovaValorePerChiave(sezione, "Origin");
                const arrivoEl = trovaValorePerChiave(sezione, "Sched. Arrival");
                const actualArrivoEl = trovaValorePerChiave(sezione, "Actual Arrival");
                let risultato = null;
                let codiceOrigine = null;
                let testoArrivo = "";
                let testoActualArrivo = "";
                let testoArrivoPorta = "";
                if (origineEl && arrivoEl) {
                    codiceOrigine = estraiCodiceGrezzo(origineEl);
                    testoArrivo = arrivoEl.textContent.trim();
                    if (codiceOrigine && !testoArrivo.toUpperCase().includes("N/A")) {
                        let arrivo = parseDataOraSlash(testoArrivo);
                        if (actualArrivoEl) {
                            testoActualArrivo = actualArrivoEl.textContent.trim();
                            if (testoActualArrivo && !testoActualArrivo.toUpperCase().includes("N/A")) {
                                const arrivoReale = parseDataOraSlash(testoActualArrivo);
                                if (arrivoReale) arrivo = arrivoReale;
                            }
                        }
                        let arrivoPortaReale = null;
                        if (baiaSelezionata) {
                            const arriviSalvati = caricaArriviBaia();
                            const voceBaia = arriviSalvati[baiaSelezionata];
                            if (voceBaia && voceBaia.orario) {
                                const d = new Date(voceBaia.orario);
                                if (!isNaN(d.getTime())) { arrivoPortaReale = d; testoArrivoPorta = voceBaia.orario; }
                            }
                        }
                        if (arrivo) risultato = calcolaCutoffPerCodice(codiceOrigine, arrivo, arrivoPortaReale);
                    }
                }
                aggiornaRigaArrivoECountdown(contenitore, risultato);
                if (risultato && baiaSelezionata) cacheCutoffPerBaia.set(baiaSelezionata, risultato);
                if (!origineEl || !arrivoEl) return;
                const chiave = codiceOrigine + "|" + testoArrivo + "|" + testoActualArrivo + "|" + testoArrivoPorta;
                if (sezione.getAttribute("data-tnt-turno-key") === chiave) return;
                sezione.setAttribute("data-tnt-turno-key", chiave);
                const vecchia = sezione.querySelector(".tnt-turno-row");
                if (vecchia) vecchia.remove();
                if (!codiceOrigine || testoArrivo.toUpperCase().includes("N/A")) return;
                const riga = document.createElement("div");
                riga.className = "row door-info__movmt-section-row tnt-turno-row";
                if (risultato) {
                    riga.appendChild(creaColTurno("Turno", risultato.cat.turno));
                    riga.appendChild(creaColTurno("Tipo Cutoff", risultato.cat.nome));
                    const hh = String(risultato.cutoffDate.getHours()).padStart(2, "0");
                    const mm = String(risultato.cutoffDate.getMinutes()).padStart(2, "0");
                    let testoCutoff = hh + ":" + mm;
                    if (risultato.ritardo && risultato.cutoffOriginale) {
                        const hhO = String(risultato.cutoffOriginale.getHours()).padStart(2, "0");
                        const mmO = String(risultato.cutoffOriginale.getMinutes()).padStart(2, "0");
                        testoCutoff = "* " + hhO + ":" + mmO + " -> " + hh + ":" + mm + " per late arrival";
                    }
                    const colCutoff = creaColTurno("Orario Cutoff", testoCutoff, risultato.ritardo ? "col-4" : "col-4");
                    if (risultato.ritardo) {
                        const valEl = colCutoff.querySelector(".door-info__movmt-value");
                        if (valEl) { valEl.style.color = "#ff3b30"; valEl.title = "Cutoff slittato: meno di 150 minuti di lavorazione tra l'arrivo reale alla baia e il cutoff originale"; }
                    }
                    riga.appendChild(colCutoff);
                } else {
                    riga.appendChild(creaColTurno("Turno", "N/A"));
                    riga.appendChild(creaColTurno("Tipo Cutoff", "N/A"));
                    riga.appendChild(creaColTurno("Orario Cutoff", "N/A"));
                }
                sezione.appendChild(riga);
            } catch (e) {}
        });
    }

    const COLORI_UNIT_PRONTA = ["rgb(135,185,172)"];

    function rigaSegnalataPronta(riga) {
        if (riga.querySelector(".unit_ready_outline") || riga.classList.contains("unit_ready_outline")) return true;
        const dockingBg = riga.querySelector('[id^="dockingBg_"]');
        if (!dockingBg) return false;
        if (dockingBg.className && typeof dockingBg.className === "string" && /ready/i.test(dockingBg.className)) return true;
        let colore = dockingBg.style && dockingBg.style.backgroundColor ? dockingBg.style.backgroundColor : "";
        if (!colore && window.getComputedStyle) {
            try { colore = window.getComputedStyle(dockingBg).backgroundColor || ""; } catch (e) {}
        }
        colore = colore.replace(/\s+/g, "");
        if (!colore) return false;
        return COLORI_UNIT_PRONTA.some((c) => colore.indexOf(c) !== -1);
    }

    // Il camion "blu" (pronto, da spostare) è marcato sull'elemento della mappa baie
    // (classe "unit_ready_outline" sul blocco, oppure "outbound_blue_icon" sull'iconcina
    // interna), NON sulla riga della lista scheduled. Va quindi cercato l'elemento mappa
    // corrispondente al numero baia, non la riga stessa.
    function camionProntoBlu(doorNumber) {
        if (!doorNumber) return false;
        const wrapper = trovaElementoBaiaPerNumero(doorNumber);
        if (!wrapper) return false;
        if (wrapper.classList && wrapper.classList.contains("unit_ready_outline")) return true;
        if (wrapper.querySelector(".unit_ready_outline")) return true;
        if (wrapper.querySelector(".outbound_blue_icon")) return true;
        return false;
    }

    function leggiOutboundInScadenza() {
        const righe = document.querySelectorAll(".scheduled-root.outbound-task, .scheduled-root.highlight-task, .scheduled-root.highlight-undeparted-unit");
        const risultati = [];
        const ora = new Date();
        righe.forEach((riga) => {
            const doorEl = riga.querySelector(".outbound-door-number span");
            const dataEl = riga.querySelector(".outbound-dates-position-change");
            const destEl = riga.querySelector(".outbound-current-destination");
            if (!doorEl || !dataEl) return;
            const door = doorEl.textContent.trim();
            const testo = dataEl.textContent.trim();
            if (!testo || testo.toUpperCase().includes("N/A") || door === "-" || !door || categoriaBaia(door) === "Scarico") return;
            if (rigaSegnalataPronta(riga) || camionProntoBlu(door)) return;
            const match = testo.match(/(\d{2})-(\d{2})\s+(\d{2}):(\d{2})/);
            if (!match) return;
            const giorno = parseInt(match[1], 10), mese = parseInt(match[2], 10), ore = parseInt(match[3], 10), minuti = parseInt(match[4], 10);
            const partenza = new Date(ora.getFullYear(), mese - 1, giorno, ore, minuti);
            const pronto = new Date(partenza.getTime() - 20 * 60000);
            const diffMinuti = (pronto.getTime() - ora.getTime()) / 60000;
            if (diffMinuti <= SOGLIA_OUTBOUND_MINUTI && diffMinuti > -60) {
                risultati.push({ door, destinazione: destEl ? destEl.textContent.trim() : "", diffMinuti, categoria: categoriaBaia(door) });
            }
        });
        risultati.sort((a, b) => a.diffMinuti - b.diffMinuti);
        const visti = new Set();
        return risultati.filter((r) => {
            const chiave = r.door + "|" + r.destinazione;
            if (visti.has(chiave)) return false;
            visti.add(chiave);
            return true;
        });
    }

    const GIORNI_CONSERVAZIONE_STORICO = 3;

    function pulisciStoricoVecchio() {
        const storico = caricaStoricoReport();
        if (storico.length === 0) return;
        const limite = Date.now() - GIORNI_CONSERVAZIONE_STORICO * 24 * 60 * 60 * 1000;
        const filtrato = storico.filter((r) => {
            const t = r.timestampIso ? new Date(r.timestampIso).getTime() : NaN;
            return !isNaN(t) && t >= limite;
        });
        if (filtrato.length !== storico.length) salvaStoricoReport(filtrato);
    }

    function registraInStoricoReport(doorNumber, unitId, orarioIso) {
        pulisciStoricoVecchio();
        const storico = caricaStoricoReport();
        const esisteGia = storico.some(item => item.doorNumber === doorNumber && item.unitId === unitId);
        if (esisteGia) return;

        const dataArrivo = new Date(orarioIso);
        const orarioFormattato = String(dataArrivo.getHours()).padStart(2, "0") + ":" + String(dataArrivo.getMinutes()).padStart(2, "0");
        const dataFormattata = String(dataArrivo.getDate()).padStart(2, "0") + "/" + String(dataArrivo.getMonth() + 1).padStart(2, "0");

        let codiceOrigine = "N/A";
        let localita = "N/A";
        let viaggio = "N/A";
        let cutoffOrario = "N/A";
        let cutoffTurno = "N/A";
        let cutoffTipo = "N/A";
        let ritardoFlag = "No";
        let cutoffOriginaleTesto = "";

        if (unitId) {
            const mov = cacheMovimentiPerUnit.get(String(unitId));
            if (mov) {
                codiceOrigine = mov.origin || "N/A";
                localita = codeMap[codiceOrigine] || "N/A";
                viaggio = mov.movementName || "N/A";
                if (mov.tripTourDescription) viaggio += " (" + mov.tripTourDescription + ")";

                const arrivoIso = mov.actualArrival || mov.scheduleArrival || mov.estimatedArrival;
                if (arrivoIso) {
                    const arrivoDate = new Date(arrivoIso);
                    const arrivoPortaReale = new Date(orarioIso);
                    const res = calcolaCutoffPerCodice(codiceOrigine, arrivoDate, !isNaN(arrivoPortaReale.getTime()) ? arrivoPortaReale : null);
                    if (res) {
                        cutoffOrario = String(res.cutoffDate.getHours()).padStart(2, "0") + ":" + String(res.cutoffDate.getMinutes()).padStart(2, "0");
                        cutoffTurno = res.cat.turno;
                        cutoffTipo = res.cat.nome;
                        if (res.ritardo && res.cutoffOriginale) {
                            ritardoFlag = "Si";
                            cutoffOriginaleTesto = String(res.cutoffOriginale.getHours()).padStart(2, "0") + ":" + String(res.cutoffOriginale.getMinutes()).padStart(2, "0");
                        }
                    }
                }
            }
        }

        storico.push({
            data: dataFormattata,
            timestampIso: orarioIso,
            doorNumber,
            unitId: unitId || "N/A",
            orarioArrivo: orarioFormattato,
            codiceOrigine,
            localita,
            viaggio,
            cutoffOrario,
            cutoffTurno,
            cutoffTipo,
            ritardoFlag,
            cutoffOriginaleTesto
        });
        salvaStoricoReport(storico);
    }

    function scaricaReportCSV(dataInizioStr, dataFineStr) {
        let storico = caricaStoricoReport();
        if (storico.length === 0) {
            alert("Nessun dato registrato per il report.");
            return;
        }

        if (dataInizioStr || dataFineStr) {
            const inizio = dataInizioStr ? new Date(dataInizioStr + "T00:00:00") : null;
            const fine = dataFineStr ? new Date(dataFineStr + "T23:59:59") : null;
            storico = storico.filter((r) => {
                if (!r.timestampIso) return true;
                const d = new Date(r.timestampIso);
                if (isNaN(d.getTime())) return true;
                if (inizio && d < inizio) return false;
                if (fine && d > fine) return false;
                return true;
            });
        }

        if (storico.length === 0) {
            alert("Nessun dato registrato nel periodo selezionato.");
            return;
        }

        let csvContent = "data:text/csv;charset=utf-8,";
        csvContent += "Data;Baia;Unit ID;Ora Arrivo Reale;Codice;Localita;Info Viaggio;Ora Cutoff;Turno;Categoria Cutoff;Slittato per Late Arrival;Cutoff Originale\n";

        storico.forEach(r => {
            const riga = `${r.data};${r.doorNumber};${r.unitId};${r.orarioArrivo};${r.codiceOrigine};${r.localita};${r.viaggio};${r.cutoffOrario};${r.cutoffTurno};${r.cutoffTipo};${r.ritardoFlag || "No"};${r.cutoffOriginaleTesto || ""}`;
            csvContent += riga + "\n";
        });

        const encodedUri = encodeURI(csvContent);
        const link = document.createElement("a");
        link.setAttribute("href", encodedUri);
        const suffisso = (dataInizioStr || dataFineStr)
            ? `_${dataInizioStr || "inizio"}_a_${dataFineStr || "fine"}`
            : `_TUTTO_${new Date().toISOString().slice(0,10)}`;
        link.setAttribute("download", `REPORT_TURNO_YOS${suffisso}.csv`);
        document.body.appendChild(link);
        link.click();
        document.body.removeChild(link);
    }

    function aggiornaTracciamentoArrivi() {
        const arrivi = caricaArriviBaia();
        let modificato = false;
        document.querySelectorAll(DOOR_NUM_SELECTOR_SCARICO).forEach((numEl) => {
            const span = numEl.querySelector("span");
            if (!span) return;
            const doorNumber = span.textContent.trim();
            if (!doorNumber || categoriaBaia(doorNumber) !== "Scarico") return;
            const wrapper = numEl.closest('[id^="unit_"]');
            if (!wrapper) return;
            const container = wrapper.querySelector('[id^="container_"].unit-assigned');
            if (container) {
                assenzeConsecutive.set(doorNumber, 0);
                if (!arrivi[doorNumber]) {
                    arrivi[doorNumber] = { unitId: null, orario: new Date().toISOString() };
                    modificato = true;
                }
            } else {
                const conteggio = (assenzeConsecutive.get(doorNumber) || 0) + 1;
                assenzeConsecutive.set(doorNumber, conteggio);
                if (conteggio >= 2 && arrivi[doorNumber]) {
                    delete arrivi[doorNumber];
                    modificato = true;
                }
            }
        });
        if (modificato) salvaArriviBaia(arrivi);
    }

    function aggiornaRigaArrivoECountdown(ambito, risultato) {
        const vecchia = ambito.querySelector(".tnt-custom-info-row");
        if (vecchia) vecchia.remove();
        const rigaRemarks = trovaRigaPerChiavi(ambito, ["Arrival Remarks", "YOS Remarks"]);
        if (!rigaRemarks) return;
        let valoreArrivo = "N/A";
        if (baiaSelezionata) {
            const arrivi = caricaArriviBaia();
            const voce = arrivi[baiaSelezionata];
            if (voce && voce.orario) {
                const dataArrivo = new Date(voce.orario);
                valoreArrivo = String(dataArrivo.getHours()).padStart(2, "0") + ":" + String(dataArrivo.getMinutes()).padStart(2, "0");
            }
        }
        let valoreCountdown = "N/A";
        if (risultato && risultato.cutoffDate) {
            const diffMs = risultato.cutoffDate.getTime() - new Date().getTime();
            if (diffMs > 0) {
                const diffMinsTot = Math.floor(diffMs / 60000);
                const ore = Math.floor(diffMinsTot / 60);
                const minuti = diffMinsTot % 60;
                valoreCountdown = ore > 0 ? `${ore}h ${minuti}m` : `${minuti}m`;
            } else {
                const diffMinsTot = Math.floor(Math.abs(diffMs) / 60000);
                const ore = Math.floor(diffMinsTot / 60);
                const minuti = diffMinsTot % 60;
                valoreCountdown = `In ritardo di ${ore > 0 ? ore + 'h ' : ''}${minuti}m`;
            }
        }
        const rigaNuova = document.createElement("div");
        rigaNuova.className = "row door-info__movmt-section-row tnt-custom-info-row";
        rigaNuova.style.marginTop = "10px";
        const colArrivo = creaColTurno("Orario Arrivo Camion", valoreArrivo, "col-6");
        const colCountdown = creaColTurno("Countdown Cutoff", valoreCountdown, "col-6");
        const valSpan = colCountdown.querySelector(".door-info__movmt-value");
        if (valSpan && risultato && risultato.cutoffDate) {
            const diffMs = risultato.cutoffDate.getTime() - new Date().getTime();
            if (diffMs < 0) valSpan.style.color = "#ff3b30";
            else if (diffMs <= 30 * 60000) valSpan.style.color = "#ffcc00";
            else valSpan.style.color = "#4cd964";
        }
        rigaNuova.appendChild(colArrivo);
        rigaNuova.appendChild(colCountdown);
        rigaRemarks.insertAdjacentElement("afterend", rigaNuova);
    }

    function leggiCutoffBaiaCorrente() {
        const originClassEl = document.querySelector(".inbound-origin-en-route, .inbound-origin");
        let codiceOrigine = null;
        if (originClassEl && originClassEl.textContent.trim()) codiceOrigine = estraiCodiceGrezzo(originClassEl);
        const contenitori = document.querySelectorAll(".door-info__movmt-no-left-padding");
        for (const contenitore of contenitori) {
            const sezione = contenitore.querySelector(".door-info__movmt-section");
            if (!sezione) continue;
            if (!codiceOrigine) codiceOrigine = estraiCodiceGrezzo(trovaValorePerChiave(sezione, "Origin"));
            const arrivoEl = trovaValorePerChiave(sezione, "Sched. Arrival");
            if (!arrivoEl || !codiceOrigine) continue;
            const testoArrivo = arrivoEl.textContent.trim();
            if (testoArrivo.toUpperCase().includes("N/A")) continue;
            let arrivo = parseDataOraSlash(testoArrivo);
            if (!arrivo) continue;
            const actualArrivoEl = trovaValorePerChiave(sezione, "Actual Arrival");
            if (actualArrivoEl) {
                const testoActual = actualArrivoEl.textContent.trim();
                if (testoActual && !testoActual.toUpperCase().includes("N/A")) {
                    const arrivoReale = parseDataOraSlash(testoActual);
                    if (arrivoReale) arrivo = arrivoReale;
                }
            }
            let arrivoPortaReale = null;
            if (baiaSelezionata) {
                const arriviSalvati = caricaArriviBaia();
                const voceBaia = arriviSalvati[baiaSelezionata];
                if (voceBaia && voceBaia.orario) {
                    const d = new Date(voceBaia.orario);
                    if (!isNaN(d.getTime())) arrivoPortaReale = d;
                }
            }
            const cutoffRes = calcolaCutoffPerCodice(codiceOrigine, arrivo, arrivoPortaReale);
            if (cutoffRes) return { cutoff: cutoffRes, codice: codiceOrigine };
        }
        return null;
    }

    function scansionaBaieScarico(panelBody) {
        const gruppi = new Map();
        const ora = new Date();
        let trovatoQualcosa = false;
        function aggiungiAlGruppo(risultato, infoBaia) {
            const diffMinuti = (risultato.cutoffDate.getTime() - ora.getTime()) / 60000;
            if (diffMinuti > SOGLIA_SCARICO_MINUTI) return;
            const senzaBaiaAssegnata = /^\(.*\)$/.test(infoBaia);
            // Un camion senza baia assegnata e senza orario di arrivo reale (arrivoPortaReale
            // sconosciuto) non può MAI essere marcato "in ritardo" da calcolaCutoffPerCodice
            // (quel calcolo richiede l'orario reale), quindi resta sul suo cutoff base/originale
            // (es. 21:30 Import di Prima). Va bene così: quando non sappiamo se è in ritardo,
            // non dobbiamo presumerlo e farlo sparire. Il filtro sotto va quindi applicato solo
            // per nascondere rumore di camion ancora lontani (>150min, non urgenti); non deve
            // MAI nascondere un camion già scaduto/vicino al cutoff (diffMinuti < 150 incluso
            // negativo = già in ritardo), perché è esattamente il caso più critico da mostrare.
            if (senzaBaiaAssegnata && diffMinuti < 150 && diffMinuti >= 0) return;
            const hh = String(risultato.cutoffDate.getHours()).padStart(2, "0");
            const mm = String(risultato.cutoffDate.getMinutes()).padStart(2, "0");
            const chiave = risultato.cat.nome + "|" + hh + ":" + mm;
            if (!gruppi.has(chiave)) gruppi.set(chiave, { nome: risultato.cat.nome, turno: risultato.cat.turno, orario: risultato.cutoffDate, baie: [] });
            gruppi.get(chiave).baie.push(risultato.ritardo ? infoBaia + "*" : infoBaia);
        }
        // --- 1) Baie di scarico: unione di due fonti.
        // (a) Cache persistente "arrivi baia", alimentata dall'intercettazione delle chiamate
        //     dockingDoors (ed eventualmente dal fallback DOM periodico di aggiornaTracciamentoArrivi):
        //     garantisce di non perdere un camion anche se in questo istante il DOM non lo mostra.
        // (b) Scansione diretta del DOM delle baie attualmente occupate: garantisce che se la voce
        //     in cache viene rimossa per qualsiasi motivo (es. l'API segnala momentaneamente la baia
        //     non più occupata) ma il camion è ancora fisicamente lì, non sparisca comunque dal pannello.
        const arriviBaiaCache = caricaArriviBaia();
        const doorsGiaContate = new Set();
        const movementIdGiaContati = new Set();

        function elaboraBaiaOccupata(doorNumber) {
            if (doorsGiaContate.has(doorNumber)) return;
            doorsGiaContate.add(doorNumber);
            trovatoQualcosa = true;
            const voce = arriviBaiaCache[doorNumber];
            if (voce && voce.unitId) {
                const mov = cacheMovimentiPerUnit.get(String(voce.unitId));
                if (mov && mov.movementId) movementIdGiaContati.add(mov.movementId);
            }
            let infoCutoff = calcolaCutoffDaCacheMovimenti(doorNumber);
            if (!infoCutoff) {
                const cutoffCachato = cacheCutoffPerBaia.get(doorNumber);
                if (cutoffCachato) infoCutoff = { cutoff: cutoffCachato, codice: null };
            }
            if (!infoCutoff && baiaSelezionata === doorNumber) {
                assicuraTabInbound();
                infoCutoff = leggiCutoffBaiaCorrente();
                if (infoCutoff) cacheCutoffPerBaia.set(doorNumber, infoCutoff.cutoff);
            }
            if (!infoCutoff) return;
            aggiungiAlGruppo(infoCutoff.cutoff, doorNumber);
        }

        Object.keys(arriviBaiaCache).forEach((doorNumber) => {
            if (categoriaBaia(doorNumber) !== "Scarico") return;
            elaboraBaiaOccupata(doorNumber);
        });

        document.querySelectorAll(DOOR_NUM_SELECTOR_SCARICO).forEach((numEl) => {
            const span = numEl.querySelector("span");
            if (!span) return;
            const doorNumber = span.textContent.trim();
            if (!doorNumber || categoriaBaia(doorNumber) !== "Scarico") return;
            const wrapper = numEl.closest('[id^="unit_"]');
            if (!wrapper) return;
            const container = wrapper.querySelector('[id^="container_"].unit-assigned');
            if (!container) return;
            elaboraBaiaOccupata(doorNumber);
        });

        // --- 2) Menù "scheduled": camion non ancora arrivati a una baia. Se un camion risulta
        // già contato al punto 1 (stesso movementId), viene saltato per non contarlo due volte.
        const scheduledElements = document.querySelectorAll(".schedule_units");
        if (scheduledElements.length > 0) trovatoQualcosa = true;
        const movimentiScheduledVisti = new Set();
        scheduledElements.forEach((el) => {
            const idEl = el.querySelector(".scheduled-movement-id");
            const originEl = el.querySelector(".scheduled-orgin");
            const arrivalEl = el.querySelector(".scheduled-sta");
            if (!originEl || !arrivalEl) return;
            const movementId = idEl ? idEl.textContent.trim() : null;
            if (movementId) {
                if (movimentiScheduledVisti.has(movementId)) return;
                movimentiScheduledVisti.add(movementId);
                if (movementIdGiaContati.has(movementId)) return;
            }
            const codiceOrigine = estraiCodiceGrezzo(originEl);
            const testoArrivo = arrivalEl.textContent.trim();
            if (!codiceOrigine || !testoArrivo || testoArrivo.toUpperCase().includes("N/A")) return;
            const arrivo = parseDataOraSlash(testoArrivo);
            if (!arrivo) return;
            const risultato = calcolaCutoffPerCodice(codiceOrigine, arrivo);
            if (!risultato) return;
            aggiungiAlGruppo(risultato, "(" + codiceOrigine + ")");
        });

        // --- 3) "En-route": camion in viaggio verso una baia specifica. Se quella baia è già
        // stata contata al punto 1, viene saltato per non duplicare lo stesso camion.
        const unitElements = document.querySelectorAll(".arrival_units");
        if (unitElements.length > 0) trovatoQualcosa = true;
        unitElements.forEach((el) => {
            const originEl = el.querySelector(".inbound-origin-en-route, .inbound-origin");
            const arrivalEl = el.querySelector(".arrival-time-en-route, .arrival-time");
            if (!originEl || !arrivalEl) return;
            const codiceOrigine = estraiCodiceGrezzo(originEl);
            const testoArrivo = arrivalEl.textContent.trim();
            if (!codiceOrigine || testoArrivo.toUpperCase().includes("N/A")) return;
            const arrivo = parseDataOraSlash(testoArrivo);
            if (!arrivo) return;
            const doorEl = el.querySelector(".door_number, .west_door_number, .east_door_number, .reverse_door_number");
            const doorNumberTesto = doorEl ? doorEl.textContent.trim() : null;
            if (doorNumberTesto && doorsGiaContate.has(doorNumberTesto)) return;
            let arrivoPortaReale = null;
            if (doorNumberTesto) {
                const voceBaia = arriviBaiaCache[doorNumberTesto];
                if (voceBaia && voceBaia.orario) {
                    const d = new Date(voceBaia.orario);
                    if (!isNaN(d.getTime())) arrivoPortaReale = d;
                }
            }
            const risultato = calcolaCutoffPerCodice(codiceOrigine, arrivo, arrivoPortaReale);
            if (!risultato) return;
            aggiungiAlGruppo(risultato, doorNumberTesto || "(" + codiceOrigine + ")");
        });

        if (!trovatoQualcosa) {
            panelBody.innerHTML = '<div class="tnt-empty">Nessun camion/baia trovato</div>';
            return;
        }
        renderPannelloScaricoConteggi(panelBody, gruppi);
    }

    function renderPannelloScaricoConteggi(panelBody, gruppi) {
        if (gruppi.size === 0) {
            panelBody.innerHTML = '<div class="tnt-empty">Nessuno scarico in scadenza</div>';
            return;
        }
        const ora = new Date();
        const voci = [...gruppi.entries()].sort((a, b) => a[1].orario.getTime() - b[1].orario.getTime());
        const fasce = [
            { chiave: "ritardo", etichetta: "In ritardo", classe: "tnt-card-rosso", test: (m) => m < 0 },
            { chiave: "30", etichetta: "Entro 30 min", classe: "tnt-card-rosso", test: (m) => m <= 30 },
            { chiave: "60", etichetta: "Entro 60 min", classe: "tnt-card-ambra", test: (m) => m <= 60 },
            { chiave: "oltre", etichetta: "Successivi", classe: "tnt-card-ambra", test: () => true }
        ];
        const bucket = new Map();
        voci.forEach(([chiave, info]) => {
            const diffMinuti = (info.orario.getTime() - ora.getTime()) / 60000;
            const fascia = fasce.find((f) => f.test(diffMinuti));
            if (!bucket.has(fascia.chiave)) bucket.set(fascia.chiave, { fascia, voci: [] });
            bucket.get(fascia.chiave).voci.push(info);
        });
        let html = "";
        let presenteAsterisco = false;
        fasce.forEach((f) => {
            const gruppo = bucket.get(f.chiave);
            if (!gruppo || gruppo.voci.length === 0) return;
            html += '<div class="tnt-group-label">' + f.etichetta + '</div>';
            gruppo.voci.forEach((info) => {
                const hh = String(info.orario.getHours()).padStart(2, "0");
                const mm = String(info.orario.getMinutes()).padStart(2, "0");
                const listaBaie = info.baie.filter(Boolean).join(", ");
                if (listaBaie.indexOf("*") !== -1) presenteAsterisco = true;
                html += '<div class="tnt-card ' + f.classe + '">'
                    + '<div class="tnt-card-row">'
                    + '<span class="tnt-card-title">' + info.nome + ' (' + info.baie.length + ')</span>'
                    + '<span class="tnt-badge">' + hh + ':' + mm + '</span>'
                    + '</div>'
                    + '<div class="tnt-card-sub">Baie: ' + listaBaie + '</div>'
                    + '</div>';
            });
        });
        if (presenteAsterisco) {
            html += '<div class="tnt-footnote">* slittato al cutoff successivo per late arrival (meno di 150min di lavorazione disponibili)</div>';
        }
        panelBody.innerHTML = html;
    }

    function aggiornaPannelloOutbound() {
        const body = document.getElementById("tnt-orologio-panel-body");
        const panel = document.getElementById("tnt-orologio-panel");
        if (!body || !panel || panel.style.display === "none") return;
        const risultati = leggiOutboundInScadenza();
        const ordineCategorie = ["Carico", "Parcheggio", "Freight"];
        const gruppi = { "Carico": [], "Parcheggio": [], "Freight": [] };
        risultati.forEach((r) => { if(gruppi[r.categoria]) gruppi[r.categoria].push(r); });
        let html = "";
        ordineCategorie.forEach((cat) => {
            const lista = gruppi[cat];
            if (!lista || lista.length === 0) return;
            html += '<div class="tnt-group-label">' + cat + ' (' + lista.length + ')</div>';
            html += lista.map((r) => {
                const minutiTesto = r.diffMinuti >= 0 ? Math.round(r.diffMinuti) + " min" : "In ritardo";
                const classeCard = r.diffMinuti <= 30 ? "tnt-card-rosso" : "tnt-card-ambra";
                return '<div class="tnt-card ' + classeCard + '">'
                    + '<div class="tnt-card-row">'
                    + '<span class="tnt-card-title">' + r.door + (r.destinazione ? " · " + r.destinazione : "") + '</span>'
                    + '<span class="tnt-badge">' + minutiTesto + '</span>'
                    + '</div>'
                    + '</div>';
            }).join("");
        });
        body.innerHTML = html || '<div class="tnt-empty">Nessuna baia in scadenza</div>';
    }

    function chiudiPannelliEsclusi(idsDaMantenere) {
        document.querySelectorAll(".tnt-panel").forEach((p) => {

    function pannelloAperto(panel) {
        return !!panel && panel.style.display !== "none" && panel.style.display !== "";
    }

    const ORDINE_GRUPPO_CARICO_SCARICO = ["tnt-orologio-panel", "tnt-orologio-scarico-panel"];

    function posizionaPannelliGruppo(ordineIds) {
        const menu = document.querySelector(".cc-hub-menu-panel");
        if (!menu) return;
        const rectMenu = menu.getBoundingClientRect();
        const altezzaDisponibile = window.innerHeight - rectMenu.top - 12;
        let leftCorrente = rectMenu.right;
        ordineIds.forEach((id) => {
            const p = document.getElementById(id);
            if (!pannelloAperto(p)) return;
            p.style.top = rectMenu.top + "px";
            p.style.left = leftCorrente + "px";
            p.style.height = Math.max(altezzaDisponibile, 200) + "px";
            leftCorrente += p.getBoundingClientRect().width;
        });
    }

    function posizionaPannelloSingolo(panel, btn) {
        const menu = document.querySelector(".cc-hub-menu-panel");
        const rectMenu = menu ? menu.getBoundingClientRect() : btn.getBoundingClientRect();
        const altezzaDisponibile = window.innerHeight - rectMenu.top - 12;
        panel.style.top = rectMenu.top + "px";
        panel.style.left = rectMenu.right + "px";
        panel.style.height = Math.max(altezzaDisponibile, 200) + "px";
    }

    function collegaClick(btn, panel, callbackAggiorna, gruppoIds) {
        btn.addEventListener("click", (e) => {
            e.stopPropagation();
            const eraAperto = pannelloAperto(panel);
            if (gruppoIds) {
                chiudiPannelliEsclusi(gruppoIds);
                panel.style.display = eraAperto ? "none" : "flex";
                posizionaPannelliGruppo(gruppoIds);
                if (!eraAperto) callbackAggiorna();
                return;
            }
            chiudiTuttiIPannelli();
            if (eraAperto) return;
            panel.style.display = "flex";
            posizionaPannelloSingolo(panel, btn);
            callbackAggiorna();
        });
        panel.addEventListener("click", (e) => e.stopPropagation());
    }

    let debounceTimer = null;
    const nodiDaControllare = new Set();
    function eseguiControlliPesanti() {
        const nodi = Array.from(nodiDaControllare);
        nodiDaControllare.clear();
        try { nodi.forEach((n) => { if (document.body.contains(n)) scanBlocchi(n); }); } catch (e) {}
        try { initIcone(); } catch (e) {}
        try { aggiornaTracciamentoArrivi(); } catch (e) {}
        try { processaTripDetails(); } catch (e) {}
        try { aggiornaPannelloOutbound(); } catch (e) {}
    }

    const observer = new MutationObserver((mutations) => {
        for (const m of mutations) {
            if (m.type === "childList") {
                m.addedNodes.forEach((n) => { if (n.nodeType === 1) nodiDaControllare.add(n); });
            } else if (m.type === "characterData" && m.target.parentElement) {
                nodiDaControllare.add(m.target.parentElement);
            } else if (m.type === "attributes" && m.target.nodeType === 1) {
                nodiDaControllare.add(m.target);
            }
        }
        clearTimeout(debounceTimer);
        debounceTimer = setTimeout(eseguiControlliPesanti, 600);
    });
    observer.observe(document.body, { childList: true, subtree: true, characterData: true, attributes: true, attributeFilter: ["class"] });

    setInterval(aggiornaTracciamentoArrivi, 5000);
    setInterval(aggiornaPannelloOutbound, 15000);
    setInterval(pulisciStoricoVecchio, 60 * 60 * 1000);
    setInterval(elaboraCodaVerificaBaie, 4000);
